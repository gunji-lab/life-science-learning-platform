<body>

  <h1>ラボ予約</h1>


  <label>

    日付：

    <input type="date" id="date">

  </label>


  <label>

    ラボ：

    <select id="lab">

      <option value="ラボA">ラボA</option>

      <option value="ラボB">ラボB</option>

    </select>

  </label>


  <label>

    モード：

    <select id="mode">

      <option value="全室">全室</option>

      <option value="設備ごと">設備ごと</option>

    </select>

  </label>


  <label>

    設備：

    <input type="text" id="equipment" placeholder="設備名（設備ごとのとき）">

  </label>


  <label>

    時間：

    <input type="text" id="customTime" placeholder="例：10:00-12:00">

  </label>


  <button id="reserveBtn">予約する</button>


  <div id="result"></div>


  <script>

    const GAS_URL = "https://script.google.com/a/macros/toyo.jp/s/AKfycby525PtNrE1l-wXd8Eq6LZTusnFj1HJ-zXOPXClX0xRnNYr450MZJt9o48nvCQeQy6T/exec"; // ← ここをあなたのGAS URLに


    document.getElementById("reserveBtn").addEventListener("click", addReservation);


    function addReservation() {

      const selectedDate = document.getElementById("date").value;

      const selectedLab = document.getElementById("lab").value;

      const selectedMode = document.getElementById("mode").value;

      const selectedEquipment = document.getElementById("equipment").value || "全室";

      const custom = document.getElementById("customTime").value;

      const timeDisplay = custom || "時間未設定";


      if (!selectedDate) {

        alert("日付を選択してください");

        return;

      }


      const payload = {

        date: selectedDate,

        lab: selectedLab,

        mode: selectedMode,

        equipment: selectedMode === "設備ごと" ? selectedEquipment : "全室",

        time: timeDisplay

      };


      // スプシへ送信

      fetch(GAS_URL, {

        method: "POST",

        headers: { "Content-Type": "application/json" },

        body: JSON.stringify(payload)

      }).then(() => {

        document.getElementById("result").innerHTML = `

          <strong>予約を送信しました（GASにPOST済み）</strong><br><br>

          <strong>日付：</strong> ${payload.date}<br>

          <strong>ラボ：</strong> ${payload.lab}<br>

          <strong>モード：</strong> ${payload.mode}<br>

          <strong>設備：</strong> ${payload.equipment}<br>

          <strong>時間：</strong> ${payload.time}<br>

        `;

      }).catch(err => {

        document.getElementById("result").innerHTML = `

          <strong>送信に失敗しました</strong><br>

          ${err}

        `;

      });

    }

  </script>

</body>
