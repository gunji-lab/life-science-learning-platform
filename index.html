<script>

let selectedLab = "";

let selectedMode = "";

let selectedEquipment = "";

let selectedTime = "";


function clearSelection(className) {

    document.querySelectorAll("." + className).forEach(btn => btn.classList.remove("selected"));

}


function selectLab(lab) {

    selectedLab = lab;

    clearSelection("lab-btn");

    event.target.classList.add("selected");

}


function selectMode(mode) {

    selectedMode = mode;

    clearSelection("mode-btn");

    event.target.classList.add("selected");

}


function selectEquipment(eq) {

    selectedEquipment = eq;

    clearSelection("equipment-btn");

    event.target.classList.add("selected");

}


function selectTime(time) {

    selectedTime = time;

    clearSelection("time-btn");

    event.target.classList.add("selected");

}


function showResult() {

    const custom = document.getElementById("customTime").value;

    const timeDisplay = custom ? custom : selectedTime;


    let text = `

        <strong>ラボ：</strong> ${selectedLab}<br>

        <strong>予約モード：</strong> ${selectedMode}<br>

    `;


    if (selectedMode === "設備ごと") {

        text += `<strong>設備：</strong> ${selectedEquipment}<br>`;

    }


    text += `<strong>時間：</strong> ${timeDisplay}<br>`;


    document.getElementById("result").innerHTML = text;

}

</script>
